1\. BATIMENTCREATE TABLE Batiment (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    nom VARCHAR(100) NOT NULL,    numero INTEGER NOT NULL UNIQUE,    allocation\_kw\_maximale REAL NOT NULL,    demande\_kw\_totale\_actuelle REAL DEFAULT 0,    capacite\_solaire\_utilisee REAL DEFAULT 0,    created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,    updated\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP);2\. NIVEAUCREATE TABLE Niveau (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    numero INTEGER NOT NULL,    batiment\_id UUID NOT NULL REFERENCES Batiment(id) ON DELETE CASCADE,    allocation\_kw\_cible REAL NOT NULL,    demande\_kw\_actuelle REAL DEFAULT 0,    demande\_kw\_precedente REAL DEFAULT 0,    delta\_demande\_kw REAL DEFAULT 0,    created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP);3\. salleCREATE TABLE Salle (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    nom VARCHAR(100) NOT NULL,    type\_salle VARCHAR(50),    niveau\_id UUID NOT NULL REFERENCES Niveau(id) ON DELETE CASCADE,    allocation\_kw\_cible REAL NOT NULL,    created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP);4\. Equipement (inféré des listes dans Batiment/Salle)CREATE TABLE Equipement (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    nom VARCHAR(100) NOT NULL,    type VARCHAR(100),    puissance\_nominale REAL NOT NULL,    est\_critique BOOLEAN DEFAULT FALSE,    allocation\_kw\_minimale\_critique REAL DEFAULT 0,    priorite INTEGER DEFAULT 0,    consommation\_actuelle REAL DEFAULT 0,    salle\_id UUID REFERENCES Salle(id) ON DELETE CASCADE,    source\_actuelle\_id UUID REFERENCES SourceEnergie(id) ON DELETE SET NULL,    created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP);5\. Relais (lié à Equipement)CREATE TABLE Relais (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    type VARCHAR(50) NOT NULL,    tension\_bobine REAL,    courant\_max REAL,    etat\_actuel BOOLEAN DEFAULT FALSE,    equipement\_id UUID NOT NULL REFERENCES Equipement(id) ON DELETE CASCADE,    date\_dernier\_changement TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,    nombre\_cycles INTEGER DEFAULT 0);6\. CapteurCREATE TABLE Capteur (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    type VARCHAR(50) NOT NULL,    unite VARCHAR(20),    valeur\_actuelle REAL,    precision REAL,    equipement\_id UUID NOT NULL REFERENCES Equipement(id) ON DELETE CASCADE,    date\_derniere\_mesure TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,    frequence\_mesure INTEGER DEFAULT 60  -- en secondes);7\. SourceEnergie (référencée dans Batiment et Equipement)CREATE TABLE SourceEnergie (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    type VARCHAR(50) NOT NULL,  -- ex: 'KIT\_SOLAIRE', 'BATTERIE', 'RESEAU\_PUBLIC'    capacite\_maximale REAL,    pourcentage\_utilise REAL DEFAULT 0,    cout\_par\_kwh REAL,    date\_derniere\_activation TIMESTAMP WITH TIME ZONE);8\. ConsommationJournaliereCREATE TABLE ConsommationJournaliere (    id UUID PRIMARY KEY DEFAULT gen\_random\_uuid(),    date DATE NOT NULL,    kwh REAL NOT NULL,    entite\_type VARCHAR(30) NOT NULL,  -- 'BATIMENT', 'NIVEAU', 'SALLE', 'EQUIPEMENT'    entite\_id UUID NOT NULL,    source\_energie\_id UUID REFERENCES SourceEnergie(id) ON DELETE SET NULL,    cout REAL,    created\_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT\_TIMESTAMP,    CONSTRAINT unique\_consommation UNIQUE (date, entite\_type, entite\_id));CREATE TABLE moteur\_analyse (    id              UUID PRIMARY KEY,    seuil\_solaire   NUMERIC (5,3) NOT NULL,    priorite        NUMERIC (5,3) NOT NULL,    id\_source\_energie       INT NOT NULL REFERENCES source\_energie (id),    id\_conf\_salle   INT NOT NULL REFERENCES conf\_salle(id));CREATE TABLE moteur (    id              UUID PRIMARY KEY,    seuil\_solaire   NUMERIC(4,3) NOT NULL,    priorite        NUMERIC(4,3) NOT NULL,    id\_source\_energie       UUID NOT NULL REFERENCES source\_energie(id),    id\_conf\_salle   UUID NOT NULL REFERENCES conf\_salle(id));**NB.**Apprendre les commandes SQL de base :CREATE TABLE, DROP TABLEINSERT INTO, SELECT, UPDATE, DELETEWHERE, ORDER BY, LIMITJOIN (INNER, LEFT, RIGHT)
